.PHONY:	all clean

all: calc clean

calc: mkDir main.o game.o
	gcc build/main.o build/game.o -o bin/100_matches

main.o: src/main.c
	gcc -Wall -Werror -c src/main.c -o build/main.o

game.o: src/game.cpp
	gcc -Wall -Werror -cpp src/game.cpp -o build/game.o

clean:
	rm -rf build/*.o

mkDir:
	mkdir bin
	mkdir build

rmDir: 
	rm -r bin
	rm -r build
